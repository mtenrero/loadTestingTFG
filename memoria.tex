\begin{lstlisting}[language=yaml, caption=ATQ docker-compose.yaml]
version: 3.3

services:
    atq:
        image: tenrero/atq
        tty: true
        volumes:
            - gluster:/gluster:rw
            - /var/run/docker.sock:/var/run/docker.sock
        ports:
            - "8080:8080"
        deployment:
            mode: global
            placement:
                constraints: [node.role == manager]
volumes:
    gluster:

\end{lstlisting}
